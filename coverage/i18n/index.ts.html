export function t(
  lang: Lang,
  key: string,
  params?: Record<string, string | number>
): string {
  const l = lang in dicts ? lang : DEFAULT_LANG;
  let s = dicts[l][key] || key;
  if (params) {
    for (const [k, v] of Object.entries(params)) {
      s = s.replace(new RegExp(`\\{${k}\\}`, "g"), String(v));
    }
  }
  return s;
}
